import socket
#echo server (в первом окне отдельного проекта)
HOST = '127.0.0.1'
PORT = 65432
with socket.socket(socket.AF_INET,socket.SOCK_STREAM) as s:
    s.bind((HOST,PORT))
    s.listen()
    conn, addr = s.accept()
    while True:
        data = conn.recv(1024)
        print(f'{data}')
        message = input()
        b = bytes(message, 'utf-8')
        conn.send(b)
