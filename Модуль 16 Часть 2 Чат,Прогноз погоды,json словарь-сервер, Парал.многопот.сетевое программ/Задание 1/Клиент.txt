import socket
HOST = '127.0.0.1'
PORT = 65432
server = HOST,PORT
with socket.socket(socket.AF_INET,socket.SOCK_STREAM) as s:
    s.connect((HOST,PORT))
    while True:
        message=input()
        b = bytes(message, 'utf-8')
        s.send(b)
        data = s.recv(1024)
        print(data)
