class checkfile:
    zakazK = 0 #Колво заказов
    MargK = 0  # Колво Маргаритa
    NeapK = 0  # Колво Неаполитанская
    cheeze4K = 0  #Колво Четыре сыра
    siccK = 0  # Колво Сицилийская
    kaprK = 0  # Колво Капричоза
    myK = 0  # Колво Моя пицца

class Piceria(checkfile):
    moneyP = 0 #безналичка
    moneyN = 0 #наличка
    zakazcheck = ''
    def __init__(self):
        pass
    def printF(self):
        with open('zakaz.txt','w') as f:
            f.writelines(f'безналичка:{self.moneyP}\n')
            f.writelines(f'наличка:{self.moneyN}\n')
            f.writelines(f'Продано:\n')
            f.writelines(f'Количество заказов:{self.zakazK}\n')
            f.writelines(f'Колво Маргаритa:{self.MargK}\n')
            f.writelines(f'Колво Неаполитанская:{self.NeapK}\n')
            f.writelines(f'Колво Четыре сыра:{self.cheeze4K}\n')
            f.writelines(f'Колво Сицилийская:{self.siccK}\n')
            f.writelines(f'Колво Капричоза:{self.kaprK}\n')
            f.writelines(f'Колво Моя пицца:{self.myK}\n')
            f.close()
